-- Table: public.service

-- DROP TABLE public.service;

CREATE TABLE public.service
(
  id integer NOT NULL DEFAULT nextval('service_id_seq'::regclass),
  name character varying NOT NULL, -- Наименование услуги
  price integer, -- Стоимость услуги
  active boolean NOT NULL DEFAULT true, -- Признак удалена ли запись или нет
  CONSTRAINT service_pkey PRIMARY KEY (id)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE public.service
  OWNER TO lwaliurowjjlfc;
COMMENT ON TABLE public.service
  IS 'Услуги';
COMMENT ON COLUMN public.service.name IS 'Наименование услуги';
COMMENT ON COLUMN public.service.price IS 'Стоимость услуги';
COMMENT ON COLUMN public.service.active IS 'Признак удалена ли запись или нет';
